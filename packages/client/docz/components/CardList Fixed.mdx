import { Playground, Props } from 'docz'
import CardDisplay from '../../src/components/CardDisplay'
import CardList from '../../src/components/CardList'
import { exampleCardModel } from '../../src/util/docz-helpers/card'
import CssBaseline from '@material-ui/core/CssBaseline';
import { createMuiTheme, MuiThemeProvider } from '@material-ui/core/styles';

# CardList Fixed
Pass pixel size to cardlist, proper transitions

<Props of={CardList} />


## Row Fixed

<Playground>
    <MuiThemeProvider theme={createMuiTheme({
        palette: {
            primary: {
            main: '#1e88e5'
            },
            secondary: {
            main: '#d81b60'
            },
            type: 'dark'
        }})}>
        <CssBaseline />
    <CardList
        direction='row'
        cellSize={{height: 301, width: 215}}
    >
        {Array.from({length: 7}).map(()=> {
            return (
                <CardDisplay
                    card={exampleCardModel()}
                    selected={false}
                    selecting={false}
                    cardHeight={32}
                    isHovered={false}
                />
            )
        })}
    </CardList>
    </MuiThemeProvider>
</Playground>
